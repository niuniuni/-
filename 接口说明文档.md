-查询首页数据
>/book/index
```
{
    item:[{
        "ad_name":内容区域,
        "data":{
            "data":数据内容
        }
    }]
}
```
-首页上拉加载
>/book/list
```
{
    items:{
        data:[{
            authors:作者,
            cover:图片,
            latest：最新章节
            fiction_id:内容id,
            title:书名
        }]
    }
}
```
-传递数据
```
{
    pagenum:页码
    limit:条数
}
```
